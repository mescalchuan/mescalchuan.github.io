    # 四、特殊数值

    null和undefined的区别 null指空值，undefined指没有值；null指曾赋过值，现在没有值，undefined值从未赋过值；null是特殊关键字，不是标识符，不能当做变量来使用，而undefined是标识符，可以当做变量使用。 ﻿ void：让表达式不返回值，但变量本身没有被更改

    var a = 2;
console.log(void a, a); //undefined 2

    //es6
Object.is(a, b)
//es5
Object.is = function(a, b) {
    if(a === 0 && b === 0) {
        return 1 / a === 1 / b //-0的情况，因为1/-为-Infinity, 0===-0为true
    }
    if(a !== a){
        return b !== b //NaN的情况
    }
    else {
        return a === b
    }
}

    # 五、值和引用

    单值：number、string、null、undefined、boolean 复合值：object（包括array）、function

    var a = [1, 2, 3];
var b = a;
b.push(4);
a; //[1, 2, 3, 4];
//改变了b的引用
b = [7, 8, 9];
b.push(10);
b; //[7, 8, 9, 10]
a; //[1, 2, 3, 4]
//只是清空数组，并没有更改引用
b = a;
b.length = 0;
b; //[]
a; //[]
