//eventLoop ->事件队列，由外部push
var eventLoop = [];
var event
while(true) {
    //一次tick
    if(eventLoop.length > 0) {
        event = eventLoop.shift();
        try { event() }
        catch(err) { reportError(err); }
    }
}

javascript是单线程的，函数具有原子性，一旦foo或bar开始运行，bar或foo的任意一行代码都必须在前一个函数执行后才会执行，也就是说异步的结果只有两种可能

a = 1, b = 2;
foo = () => {a++; b = a + b;}
bar = () => {a--; b = a - b}
